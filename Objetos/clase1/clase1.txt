object laika {
	const raza = "salchicha" 
	var energia = 100
	method ladrar() { 
		return "guau"
	}
	method salirAPasear() {
		energia = energia - 20
	} 
	method raza() { 
		return raza
	} 
	method energia() { // getter 
		return energia 
	} 
	method energia(nuevaEnergia) { // setter 
		energia = nuevaEnergia
	} 
} 

object joaco {
	var mascota = laika 
	var energia = 100
	method salirAPasear() {
		mascota.salirAPasear() 
		energia -= 20  
	} 
}
